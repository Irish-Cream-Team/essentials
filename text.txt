# Source: master-chart/templates/mongo.yaml
apiVersion: batch/v1
kind: Pod
metadata:
  name: mongo
  labels:
    helm.sh/chart: master-chart-0.2.4
    app.kubernetes.io/name: master-chart
    app.kubernetes.io/instance: essential
    app.kubernetes.io/version: "1.16.0"
    app.kubernetes.io/managed-by: Helm
  app: mongo
spec:
  replicas: 1
  selector:
    matchLabels:
      app: mongo
  template:
    metadata:
      labels:
        app: mongo
    spec:
      imagePullSecrets:
      containers:

      - name: mongo
        image: mongo:4
        command: [ "/usr/bin/mongod", "--replSet", "rs2" , "--bind_ip_all", "--wiredTigerCacheSizeGB", "1" ]
        ports:
          - containerPort: 27017
            name: mongo
          
      - name: "mongo-setup"
        image: harborreg-2.northeurope.cloudapp.azure.com/library/mongo-setup:latest
        command: ["bash", "-c", "/config/wait-for-it.sh mongo:27017 -- /config/mongo-setup.sh"]
        env:
        - name: HOST_NAME
          value: mongo
        - name: RPL_NAME
          value: rs2
        restartPolicy: OnFailure
---
